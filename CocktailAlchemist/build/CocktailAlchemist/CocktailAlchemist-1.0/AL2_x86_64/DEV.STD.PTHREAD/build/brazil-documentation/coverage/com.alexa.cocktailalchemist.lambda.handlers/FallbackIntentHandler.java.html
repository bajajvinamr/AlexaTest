package com.alexa.cocktailalchemist.lambda.handlers;

import com.alexa.cocktailalchemist.lambda.constants.RequestIntentNames;
import com.alexa.cocktailalchemist.lambda.utils.ResourceBundleProvider;
import com.amazon.ask.dispatcher.request.handler.HandlerInput;
import com.amazon.ask.dispatcher.request.handler.RequestHandler;
import com.amazon.ask.model.Response;

import java.util.Optional;
import java.util.ResourceBundle;

import static com.amazon.ask.request.Predicates.intentName;

/**
 * A basic skill handler to handle FallbackIntent. It will handle all
 * requests that weren't handled by implemented handlers.
 */
public class FallbackIntentHandler implements RequestHandler {

    /**
     * Validates whether the intent request is FallbackIntent or not.
     * @param input HandlerInput
     * @return boolean
     */
    @Override
    public boolean canHandle(HandlerInput input) {
        return input.matches(intentName(RequestIntentNames.AMAZON_FALLBACK_INTENT));
    }

    /**
     * This method sends a "Sorry, I don't know that. You can say try saying help!" message.
     * @param input HandlerInput
     * @return Response
     */
    @Override
    public Optional<Response> handle(HandlerInput input) {
        final ResourceBundle resourceBundle = ResourceBundleProvider.getResourceBundleFromHandlerInput(input);
        final String speechText = resourceBundle.getString("FALLBACK_INTENT_SPEECH");
        final String skillCardDisplayName = resourceBundle.getString("SKILL_CARD_DISPLAY_NAME");
        return input.getResponseBuilder()
                .withSpeech(speechText)
                .withSimpleCard(skillCardDisplayName, speechText)
                .withReprompt(speechText)
                .build();
    }
}
