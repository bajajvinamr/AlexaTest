package com.alexa.cocktailalchemist.lambda.handlers;

import com.alexa.cocktailalchemist.lambda.utils.ResourceBundleProvider;
import com.amazon.ask.dispatcher.request.handler.HandlerInput;
import com.amazon.ask.dispatcher.request.handler.RequestHandler;
import com.amazon.ask.model.LaunchRequest;
import com.amazon.ask.model.Response;
import com.amazon.ask.request.Predicates;

import java.util.Optional;
import java.util.ResourceBundle;

/**
 * A basic skill handler to handle launch requests from user.
 */
public class LaunchRequestHandler implements RequestHandler {
    @Override
    public boolean canHandle(HandlerInput input) {
        return input.matches(Predicates.requestType(LaunchRequest.class));
    }

    @Override
    public Optional<Response> handle(HandlerInput input) {
        String speechText = "Welcome to Cocktail Alchemist! I'm here to help you " +
                "discover delightful cocktails. You can ask for a cocktail suggestion or a recipe.";
        String repromptText = "For example, you could say, 'Suggest a gin drink.'";
        return input.getResponseBuilder()
                .withSpeech(speechText)
                .withReprompt(repromptText)
                .withShouldEndSession(false)
                .build();
    }
}
